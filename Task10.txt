
n = int(input("Введите количество монеток: "))
coins = input("Введите строку символов (O или P), где 0 - орел, 1 - решка: ")

# определяем количество монетек с гербом
count_heads = coins.count('1')
# количество монеток с орлом равно разности общего количества монеток и количества монеток с гербом
count_tails = n - count_heads

# выбираем наименьшее количество монеток, которые нужно перевернуть
if count_heads < count_tails:
    print("Минимальное количество монеток, которые нужно перевернуть: ", count_heads)
else:
    print("Минимальное количество монеток, которые нужно перевернуть: ", count_tails)
